local lib = require("prototypes.lib")

if not lib.setting("br-walls-block-spitters") then return end

local function make_spit_projectile(unit, stream)
  if not unit or not stream then return nil end
  local projectile_name = lib.safe_name("br-", unit.name, "-spit-projectile")

  local projectile = {
    type = "projectile",
    name = projectile_name,
    flags = {"not-on-map"},
    collision_box = {{-0.50, -0.25}, {0.50, 0.25}},
    hit_collision_mask = lib.collision_mask(),
    acceleration = 0,
    direction_only = true,
    hit_at_collision_position = true,
    force_condition = "not-same",
    action = lib.copy(stream.initial_action or stream.action),
    animation = lib.copy(stream.particle),
    shadow = lib.copy(stream.shadow)
  }

  lib.add_final_effects(projectile, lib.impact_effects("spit"), "br-acid-impact-smoke")
  lib.define_once("projectile", projectile)

  return projectile_name
end

for _, unit in pairs(data.raw.unit or {}) do
  local attack = unit.attack_parameters
  local delivery = attack and attack.ammo_type and attack.ammo_type.action and attack.ammo_type.action.action_delivery
  if unit.name and unit.name:find("spitter") and delivery and delivery.type == "stream" then
    local stream = data.raw.stream and data.raw.stream[delivery.stream]
    local projectile_name = make_spit_projectile(unit, stream)
    if projectile_name then
      attack.ammo_type.action.action_delivery = {
        type = "projectile",
        projectile = projectile_name,
        starting_speed = 0.3375,
        direction_deviation = 0.02,
        max_range = delivery.max_range or 35
      }
    end
  end
end
